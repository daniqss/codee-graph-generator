api/v1/
    - stats/
        - weekday/id
                URL PARAMS:
                    FROM=dd-mm-yyyy
                    TO=dd-mm-yyyy
                    TYPE="real" or "estimated"
                    SORTED="asc" or "desc" (defaults to "asc")

                RETURNS: data: {
                    cups: string, // cups del punto de suministro
                    days: int, // total de dias contados
                    total: float, // total a lo largo de todos los dias contados
                    unnits: "KWh",
                    weekday: {
                        0: {
                            totalConsumption: float,
                            meanConsumption: float,
                            medianConsumption: float,
                        }
                        1: {
                            totalConsumption: float,
                            meanConsumption: float,
                            medianConsumption: float,
                        }
                        2: ...
                        ...
                        6: ...
                    }
                }

        - day/id
                URL PARAMS: 
                    FROM=dd-mm-yyyy
                    TO=dd-mm-yyyy
                    TYPE="real" or "estimated"
                    SORTED="asc" or "desc" (defaults to "asc")

                RETURNS: data: {
                    cups: string, // cups del punto de suministro
                    days: int, // total de dias contados,
                    units: "KWh",
                    total: float, // total a lo largo del plazo
                    days: {
                        dd-mm-yyyy: {
                            consumption: float,
                        },
                        dd-mm-yyyy: {
                            consumption: float,
                        },
                        ...
                    }
                }

        - month/id
                URL PARAMS:
                    FROM=dd-mm-yyyy
                    TO=dd-mm-yyyy
                    TYPE="real" or "estimated"
                    SORTED="asc" or "desc" (defaults to "asc")
                
                RETURNS: data {
                    cups: string, // cups del punto de suministro
                    days: int, // total de dias contados,
                    units: "KWh",
                    total: float, // total a lo largo del plazo
                    months: {
                        0: {
                            totalConsumption: float,
                            meanConsumption: float,
                            medianConsumption: float,
                        },
                        1: ...
                        ...
                        11: ...
                    }
                }

        - hour/id
                URL PARAMS:
                    FROM=dd-mm-yyyy
                    TO=dd-mm-yyyy
                    TYPE="real" or "estimated"
                    SORTED="asc" or "desc" (defaults to "asc")

                RETURNS: data: {
                    cups: string, // cups del punto de suministro
                    days: int, // total de dias contados,
                    units: "KWh",
                    total: float, // total de consumo a lo largo del plazo
                    hours: {
                        0: {
                            totalConsumption: float,
                            meanConsumption: float,
                            medianConsumption: float,
                        },
                        1: {
                            ...
                        },
                        ...
                        23: ...
                    }
                }
